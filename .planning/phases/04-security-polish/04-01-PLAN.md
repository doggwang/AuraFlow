    ```javascript
    // Monitor camera permission changes
    navigator.mediaDevices.ondevicechange = async () => {
      try {
        const tracks = stream.getVideoTracks();
        if (tracks.length > 0) {
          const permState = tracks[0].readyState;
          console.log('[AuraFlow] Camera track state:', permState);
          if (permState === 'ended' || permState === 'failed') {
            this.ui.placeholder.querySelector('p').textContent = '摄像头权限已撤销';
            this.ui.placeholder.style.display = 'flex';
            this.ui.cameraDot.classList.remove('active');
          }
        }
      } catch (e) {
        console.log('[AuraFlow] Device change check failed:', e);
      }
    };
    ```
