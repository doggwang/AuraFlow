    In ParticleOcean class (around line 823, after constructor), add a debounce utility function:

    ```javascript
    // Debounce utility - delays function execution until after wait milliseconds
    debounce(func, wait) {
      let timeout;
      return function(...args) {
        clearTimeout(timeout);
        timeout = setTimeout(() => func.apply(this, args), wait);
      };
    }
    ```

    Then at line 873, change:
    ```javascript
    // Before (no debounce):
    window.addEventListener('resize', () => this.onResize());

    // After (100ms debounce):
    window.addEventListener('resize', this.debounce(() => this.onResize(), 100));
    ```
